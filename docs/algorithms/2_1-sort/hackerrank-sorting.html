import Foundation
let n = Int(readLine()!)!
let a0 = (0..<n).map { _ in return readLine()!.components(separatedBy: " ")[0]}
let ar = a0.map { return Int($0)! }

 

// count
var count = [Int: Int]()
ar.forEach { count[$0] = (count[$0] ?? 0) + 1 }

// post-count

var sum = 0
let arr : [Int] = (0..<100).map {
sum += count[$0] ?? 0
return sum
}

 // output
let str = arr.map({return "\($0)"}).joined(separator: " ")
print(str)
